BasicGame no_players=2 square_size=50
    SpriteSet
        floor > Immovable hidden=True img=oryx/backBlack
        door > Door color=GREEN img=oryx/tombstone1

        key > Resource img=oryx/skull1 limit=1 shrinkfactor=0.7

        moving >
            avatar > MovingAvatar
                avatar1 > img=oryx/reaper1
                avatar2 > img=oryx/necromancer1

        wall > Immovable img=oryx/tree2 #autotiling=True

    InteractionSet
        key avatar > collectResource

        door avatar1 > killIfOtherHasMore resource=key limit=1 scoreChange=10,0
        door avatar2 > killIfOtherHasMore resource=key limit=1 scoreChange=0,10
        avatar door > changeResource resource=key value=-1
        moving wall door > stepBack pixelPerfect=True

    TerminationSet
        SpriteCounter stype=door limit=0 count_score=True
        Timeout limit=1500 count_score=True

    LevelMapping
        d > door floor
        k > key floor
        A > avatar1 floor
        B > avatar2 floor
        . > floor
        w > floor wall
